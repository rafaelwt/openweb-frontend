src/app/core/interceptors/credentials.interceptor.ts

credentialsInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const http = inject(HttpClient); const secureReq = req.clone({ withCredentials: true, setHeaders: { 'X-Requested-With': 'XMLHttpRequest', }, }); const needsCsrf = MUTATING_METHODS.includes(req.method.toUpperCase()); const isSessionEndpoint = req.url.includes('/session'); // First mutating request: initialize CSRF if (needsCsrf && !csrfInitialized && !isSessionEndpoint) { if (!csrfInit$) { csrfInit$ = initCsrf$(http).pipe(shareReplay(1)); } return csrfInit$.pipe( switchMap(() => next(secureReq)), catchError((err) => retryOnCsrfError(err, http, secureReq, next)), ); } // Already initialized: execute with auto-retry on CSRF error if (needsCsrf && !isSessionEndpoint) { return next(secureReq).pipe(catchError((err) => retryOnCsrfError(err, http, secureReq, next))); } return next(secureReq); }

HTTP Interceptor that adds credentials and manages CSRF protection.

Features:

  • Adds withCredentials: true and X-Requested-With header to all requests
  • Lazy CSRF initialization: calls /api/session before first mutating request
  • Auto-retry on CSRF token expiration (419/403 errors)
  • Transparent token refresh without user intervention
csrfInit$
Type : Observable | null
Default value : null

Observable for CSRF initialization in progress.
csrfInitialized
Type : unknown
Default value : false

Tracks whether CSRF token has been initialized.
MUTATING_METHODS
Type : []
Default value : ['POST', 'PUT', 'DELETE', 'PATCH']

HTTP methods that require CSRF protection.
